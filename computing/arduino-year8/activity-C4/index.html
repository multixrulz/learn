Level 1 — Use a digital input

Complete the challenge

Your entire setup() function should look like this:

void setup() {
    pinMode(2, INPUT_PULLUP);    // set up pin 2 as an input.
}

Your entire setup() function should look like this:

void setup() {
    pinMode(2, INPUT_PULLUP);    // set up pin 2 as an input.
    pinMode(5, OUTPUT);          // set up pin 5 as an output.
}

or like this:

void setup() {
    pinMode(5, OUTPUT);          // set up pin 5 as an output.
    pinMode(2, INPUT_PULLUP);    // set up pin 2 as an input.
}

Your entire loop() function should look like this:

void loop() {
    if (digitalRead(2) == HIGH) {     // check if the input is on.
        digitalWrite(5, HIGH);         // turn the LED on if the input is on
    } else {
        digitalWrite(5, LOW);          // turn the LED off if the input is not on
    }
}

Level 2 — Use variables

Complete the challenge

void setup() {
    pinMode(5, OUTPUT);       // set up pin 5 as an output.
    pinMode(2, INPUT_PULLUP); // set up pin 2 as an input.
}

void loop() {
    if (digitalRead(2) == LOW) {  // check if the input is LOW
        digitalWrite(5, HIGH);    // turn the LED on
    } else {
        digitalWrite(5, LOW);     // turn the LED off
    }
}

int inputPin = 2;
int ledPin = 5;

void setup() {
    pinMode(ledPin, OUTPUT);         // set up pin "ledPin" as an output.
    pinMode(inputPin, INPUT_PULLUP); // set up pin "inputPin" as an input.
}

void loop() {
    if (digitalRead(inputPin) == LOW) {  // check if the input is LOW
        digitalWrite(ledPin, HIGH);    // turn the LED on
    } else {
        digitalWrite(ledPin, LOW);     // turn the LED off
    }
}

Level 3 — The Backwards Light

Complete the challenge

int inputPin = 3;
int ledPin = 9;

void setup() {
    pinMode(ledPin, OUTPUT);         // set up pin "ledPin" as an output.
    pinMode(inputPin, INPUT_PULLUP); // set up pin "inputPin" as an input.
}

void loop() {
    if (digitalRead(inputPin) == LOW) {  // check if the input is LOW
        digitalWrite(ledPin, HIGH);    // turn the LED on
    } else {
        digitalWrite(ledPin, LOW);     // turn the LED off
    }
}

You have 2 options. Either change something in the line that that says

    if (digitalRead(inputPin) == LOW) {  // check if the input is LOW

or change something in the lines that say

        digitalWrite(ledPin, HIGH);    // turn the LED on
    } else {
        digitalWrite(ledPin, LOW);     // turn the LED off

Make your code do the opposite of what it does now - change LOW to HIGH and change HIGH to LOW.

You have 2 options. Either change the if line to this

    if (digitalRead(inputPin) == HIGH) {  // check if the input is HIGH

or change the digitalWrite lines to

        digitalWrite(ledPin, LOW);    // turn the LED off
    } else {
        digitalWrite(ledPin, HIGH);     // turn the LED on
