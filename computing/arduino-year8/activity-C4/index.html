Level 1 — Use a digital input

Complete the challenge

Your entire setup() function should look like this:

void setup() {
    pinMode(1, INPUT);    // set up pin 1 as an input.
}

Your entire setup() function should look like this:

void setup() {
    pinMode(1, INPUT);    // set up pin 1 as an input.
    pinMode(5, OUTPUT);    // set up pin 5 as an output.
}

or like this:

void setup() {
    pinMode(5, OUTPUT);    // set up pin 5 as an output.
    pinMode(1, INPUT);    // set up pin 1 as an input.
}

Your entire loop() function should look like this:

void loop() {
    if (digitalRead(1) == HIGH) {     // check if the input is on.
        digitalWrite(5, HIGH);         // turn the LED on if the input is on
    } else {
        digitalWrite(5, LOW);          // turn the LED off if the input is not on
    }
}

Try it out

Fix the problem

Change the pinMode line for pin 1 to look like this:

    pinMode(1, INPUT_PULLUP);    // set up pin 1 as an input.

Your entire setup() function should look like this:

void setup() {
    pinMode(1, INPUT_PULLUP);    // set up pin 1 as an input.
    pinMode(5, OUTPUT);          // set up pin 5 as an output.
}

or like this:

void setup() {
    pinMode(5, OUTPUT);          // set up pin 5 as an output.
    pinMode(1, INPUT_PULLUP);    // set up pin 1 as an input.
}

Level 2 — Use variables

Complete the challenge

void setup() {
    pinMode(6, OUTPUT);       // set up pin 6 as an output.
    pinMode(4, INPUT_PULLUP); // set up pin 4 as an input.
}

void loop() {
    if (digitalRead(4) == LOW) {  // check if the input is LOW
        digitalWrite(6, HIGH);    // turn the LED on
    } else {
        digitalWrite(6, LOW);     // turn the LED off
    }
}

int inputPin = 4;
int ledPin = 6;

void setup() {
    pinMode(ledPin, OUTPUT);         // set up pin 6 as an output.
    pinMode(inputPin, INPUT_PULLUP); // set up pin 4 as an input.
}

void loop() {
    if (digitalRead(inputPin) == LOW) {  // check if the input is LOW
        digitalWrite(ledPin, HIGH);    // turn the LED on
    } else {
        digitalWrite(ledPin, LOW);     // turn the LED off
    }
}

Change the value of inputPin and ledPin.

Edit the lines int inputPin = 4; and int ledPin = 6;

Level 3 — The Backwards Light

Complete the challenge

int inputPin = 4;
int ledPin = 6;

void setup() {
    pinMode(ledPin, OUTPUT);         // set up pin 6 as an output.
    pinMode(inputPin, INPUT_PULLUP); // set up pin 4 as an input.
}

void loop() {
    if (digitalRead(inputPin) == LOW) {  // check if the input is LOW
        digitalWrite(ledPin, HIGH);    // turn the LED on
    } else {
        digitalWrite(ledPin, LOW);     // turn the LED off
    }
}

You will need to change something in the line that that says

    if (digitalRead(inputPin) == LOW) {  // check if the input is LOW

Right now the sketch checks if the input is LOW. Make it check the opposite.

Change that line to this:

    if (digitalRead(inputPin) == HIGH) {  // check if the input is HIGH
