void setup() {
    pinMode(3, OUTPUT);
    pinMode(2, INPUT_PULLUP);
}

void loop() {
    if (digitalRead(2) == LOW) {
        digitalWrite(3, HIGH);
    } else {
        digitalWrite(3, LOW);
    }
}
