Wiring an LED

Upload a sketch to your Arduino

void setup() {
    pinMode(13, OUTPUT);    // set up pin 13 as an output.
}

void loop() {
    digitalWrite(13, HIGH); // turn the LED on
    delay(1000);            // wait for 1 second
    digitalWrite(13, LOW);  // turn the LED off
    delay(1000);            // wait for 1 second
}
